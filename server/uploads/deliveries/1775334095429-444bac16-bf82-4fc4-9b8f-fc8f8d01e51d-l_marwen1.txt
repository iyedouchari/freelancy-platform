    private void connectDB() throws Exception {
        if (connect == null || connect.isClosed()) {
            Class.forName("oracle.jdbc.driver.OracleDriver");
            connect = DriverManager.getConnection(
                    "jdbc:oracle:thin:@localhost:1521:xe",
                    "system",
                    "system");
        }
    }

    private void fillFiliere() {
        isUpdating = true;
        try (Statement st = connect.createStatement();
                ResultSet rs = st.executeQuery("SELECT DISTINCT filiere FROM classe")) {

            cbFiliere.removeAllItems();
            while (rs.next()) {
                cbFiliere.addItem(rs.getString("filiere"));
            }
        } catch (SQLException e) {
            e.printStackTrace();
            JOptionPane.showMessageDialog(this, "Erreur fillFiliere: " + e.getMessage());
        } finally {
            isUpdating = false;
            if (cbFiliere.getItemCount() > 0) {
                cbFiliere.setSelectedIndex(0);
                fillMatiere(); 
            }
        }
    }

    private void fillMatiere() {
        isUpdating = true;
        try {
            cbMatiere.removeAllItems();
            String selectedFiliere = (String) cbFiliere.getSelectedItem();
           
            if (selectedFiliere != null) {
                String sql = "SELECT DISTINCT m.libelle " +
                        "FROM matiere m " +
                        "JOIN correspondance c ON m.id_matiere = c.id_matiere " +
                        "JOIN classe cl ON c.id_classe = cl.id_classe " +
                        "WHERE cl.filiere = ?";

                try (PreparedStatement ps = connect.prepareStatement(sql)) {
                    ps.setString(1, selectedFiliere);
                    try (ResultSet rs = ps.executeQuery()) {
                        while (rs.next()) {
                            cbMatiere.addItem(rs.getString("libelle"));
                        }
                    }
                }
            }

           

        } catch (SQLException e) {
            e.printStackTrace();
            JOptionPane.showMessageDialog(this, "Erreur fillMatiere: " + e.getMessage());
        } finally {
            isUpdating = false;
           
            if (cbMatiere.getItemCount() > 0) {
                cbMatiere.setSelectedIndex(0);
            } else {
                fillTable();
            }
        }
    }

    private void fillTable() {
        try {
            tableModel.setRowCount(0);
            String selectedFiliere = (String) cbFiliere.getSelectedItem();
            String selectedMatiereLibelle = (String) cbMatiere.getSelectedItem();

            
            if (selectedFiliere != null) {
             
                String idMatiere = null;
                if (selectedMatiereLibelle != null) {                	
                	PreparedStatement psMat = null;
                	ResultSet rsMat = null;
                	try {
                	    psMat = connect.prepareStatement("SELECT id_matiere FROM matiere WHERE libelle = ?");
                	    psMat.setString(1, selectedMatiereLibelle);
                	    rsMat = psMat.executeQuery();

                	    if (rsMat.next()) {
                	        idMatiere = rsMat.getString("id_matiere");
                	    }
                	} catch (SQLException e) {
                	    e.printStackTrace();
                	}
                }

                StringBuilder sql = new StringBuilder(); //String dynamique kol mara tzid fih 
                sql.append("SELECT e.id_etudiant, e.nom, e.prenom, count(a.numseance) as nb_abs ")
                        .append("FROM etudiant e ")
                        .append("JOIN classe c ON e.id_classe = c.id_classe ")
                        .append("LEFT JOIN absence a ON (e.id_etudiant = a.id_etudiant ");

                if (idMatiere != null) {
                    sql.append("AND a.id_matiere = ? ");
                }

                sql.append(") WHERE c.filiere = ? ");
               sql.append("GROUP BY e.id_etudiant, e.nom, e.prenom");


                try (PreparedStatement ps = connect.prepareStatement(sql.toString())) {
                    int idx = 1; // ken bech nestaamel 1 et 2 mahech bech temchi khater fama ifidmatiere 
                    if (idMatiere != null) {
                        ps.setString(idx++, idMatiere);
                    }
                    ps.setString(idx, selectedFiliere);

                    try (ResultSet rs = ps.executeQuery()) {
                        while (rs.next()) {
                            String id = rs.getString("id_etudiant");
                            String name = rs.getString("nom") + " " + rs.getString("prenom");
                            int abs = rs.getInt("nb_abs");
                        tableModel.addRow(new Object[] { id, name, abs });
                           
                        }
                        
                    }
                    
                }
            }
        } catch (SQLException e) {
            e.printStackTrace();
            JOptionPane.showMessageDialog(this, "Erreur fillTable SQL: " + e.getMessage());
        }
    }
